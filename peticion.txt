Solo modificar templates/cuadratura_diaria/detail.html

El botón o enlace "Volver" en el detail NO debe usar history.back().
Debe navegar directo a la lista de cuadraturas, restaurando los 
filtros guardados en sessionStorage (clave 'cuadratura_list_filters').

Reemplazar el enlace "Volver" del detail por esto:

<a href="{% url 'control:cuadratura_diaria_list' %}" 
   id="btn-volver-lista">Volver</a>

<script>
(function(){
  var a = document.getElementById('btn-volver-lista');
  if (!a) return;
  try {
    var saved = JSON.parse(sessionStorage.getItem('cuadratura_list_filters') || 'null');
    if (saved && (saved.sucursal || saved.fecha_desde || saved.fecha_hasta)) {
      var p = new URLSearchParams();
      if (saved.sucursal)    p.set('sucursal', saved.sucursal);
      if (saved.fecha_desde) p.set('fecha_desde', saved.fecha_desde);
      if (saved.fecha_hasta) p.set('fecha_hasta', saved.fecha_hasta);
      a.href = a.href.split('?')[0] + '?' + p.toString();
    }
  } catch(e) {}
})();
</script>

No tocar el botón Volver del edit (ese sí puede seguir con history.back()).
